4.Write a C program to print the following characters in a reverse way.   

___________________________________________________________________________________

#include <stdio.h> 
    int main() 
    {
       char char1 = 'X';
       char char2 = 'M';
       char char3 = 'L';
   
       printf("The reverse of %c%c%c is %c%c%c\n",
           char1, char2, char3,
           char3, char2, char1);
   
       return(0);
   }
